import fastbook
fastbook.setup_book()

from fastbook import *
from fastai.vision.widgets import *

from fastai.callback.fp16 import *

def get_data(url, presize, resize):
    path = untar_data(url)
    return DataBlock(
        blocks=(ImageBlock, CategoryBlock), get_items=get_image_files,
        splitter=GrandparentSplitter(valid_name='val'),
        get_y=parent_label, item_tfms=Resize(presize),
        batch_tfms=[*aug_transforms(min_scale=0.5, size=resize),
                    Normalize.from_stats(*imagenet_stats)],
        ).dataloaders(path, bs=128)

dls = get_data(URLs.IMAGENETTE_160, 160, 128)
dls.show_batch(max_n=4)

# resnet 34 that is not pretrained
def get_learner(**kwargs):
    return cnn_learner(dls, resnet18, pretrained=False,
    metrics=accuracy, **kwargs).to_fp16()

class PrintLoss(Callback):
    def after_epoch(self):
        print(f'Loss: {self.learn.loss}')
        return True

class LearningRateChanger(Callback):
    def __init__(self):
        self.last_losses = []

    def after_epoch(self):
        self.last_losses.append(self.learn.loss)
        print(f'Learning Rate: {self.learn.lr}')
        if (self.epoch == 0):
            print('epoch is 0')
            return True
        elif (self.last_losses[self.epoch -1] < self.learn.loss):
            print('adjusting LR')
            self.learn.lr = self.learn.lr * 0.75
        else:
            print('in else')
            print(f'current epoch: {self.epoch}')
            print(f'whole losses list: {self.last_losses}')
            print(f'last epoch loss: {self.last_losses[self.epoch -1]}')
            print(f'current loss: {self.learn.loss}')

learn = get_learner()
learn.fit_one_cycle(30, 0.003, cbs=LearningRateChanger())
